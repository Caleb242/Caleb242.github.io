big_ten_data <- basketball %>%
  filter(Conference == "Big Ten") %>%
  group_by(Team) %>%
  summarise(
    Total_Offensive_Rebounds = sum(TeamOffRebounds, na.rm = TRUE),
    Total_Rebounds = sum(TeamTotalRebounds, na.rm = TRUE), 
  ) |>
  arrange(desc(Total_Rebounds), desc(Total_Offensive_Rebounds))  # Sort by Total_Rebounds and Offensive_Rebounds
